quickbms script

```
get files short
get null short
filexor "0x78 0x6C 0x61 0x71 0x6A 0x23 0x65 0x6D 0x24 0x25 0x5E 0x66 0x6B 0x40 0x23 0x24 0x72 0x68 0x21 0x40 0x23 0x73 0x6B 0x40 0x23 0x24 0x21 0x00"
for i = 0 < files
get id long
putarray 0 i id
next i
for i = 0 < files
get id1 long
putarray 1 i id1
next i
for i = 0 < files
getarray OFFSET 1 i
if OFFSET == -1
getdstring null 0x310 #this is the recursive directory info but we can skip it in this game
endif
if OFFSET != -1
getdstring basename 0xF8
get SIZE long
get UNK01 long
get UNK02 long
getdstring NULL 0x27
getdstring NAME 0x1DD
getdstring basename 0x8
putarray 2 i SIZE
putarray 3 i NAME
endif
next i
filexor ""
savepos base
for i = 0 < files
getarray OFFSET 1 i
getarray SIZE 2 i
getarray NAME 3 i
math OFFSET + base
if NAME != 0
log NAME OFFSET SIZE
endif
next i

```
