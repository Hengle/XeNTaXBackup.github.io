		m_nVersion = VPK_CURRENT_VERSION;
		m_nDirectorySize = 0;
		m_nEmbeddedChunkSize = 0;
		m_nChunkHashesSize = 0;
		m_nSelfHashesSize = 0;
		m_nSignatureSize = 0;

};
