# Script v1.0

goto 0x100

get chunkSize long
get dummy long
get numChunks long

for i = 0 < numChunks
get name basename
string name += "_"
string name += i
string name += ".bin"

set offset i
math offset *= chunkSize
math offset += 0x800 
log name offset chunkSize 
next i
