// HEADERS (16 bytes)
// PACKED IMAGE DATA (0x30 bytes * n)
// FOOTER CHUNK 1 (0x2000 bytes, image data offsets start at +512)
// FOOTER CHUNK 2 (0x2000 bytes, empty)

DWORD EoData;     // Offset1
DWORD Offset2;    // > Offset1, FileSize - 0x2000
DWORD __unknown1; // 0
DWORD __unknown2; // Version?

struct ImgHead
{
  DWORD e;
  DWORD DXT__1, DXT__2;

  DWORD fmt;
  USHORT width, height;
  
  DWORD padding[7];

  local DWORD size = (DWORD)width * (DWORD)height;
};

local int Count = 0;

struct
{
  while( (DWORD)FTell() < EoData )
  {
    struct
    {
      ImgHead img;
      UBYTE imgData[img.size];

      Printf("Image %ux%u\n", img.width, img.height);
    } Image;

    ++Count;
  }
} ImageData;

FSeek(EoData + 512);

DWORD Offsets[ Count ];

Printf("Total of %u images\n", Count);
