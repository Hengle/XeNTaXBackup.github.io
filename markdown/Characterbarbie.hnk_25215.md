	vbuf=(F9FF8)>>76 stride 56
mesh-1:
	ibuf=(FC000)>>252
	vbuf=(FB098)>>60 stride 56
mesh-2:
	ibuf=(1381F0)>>1416
	vbuf=(FC200)>>358 stride 72
mesh-3:
	ibuf=(138D00)>>6432
	vbuf=(1026B0)>>1750 stride 72
mesh-4:
	ibuf=(13BF40)>>1134
	vbuf=(1212E0)>>245 stride 72

	ibuf=(13C81C)>>5022
